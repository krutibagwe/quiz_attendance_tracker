-- Create the database
CREATE DATABASE IF NOT EXISTS quizdatabase1;

-- Switch to the created database
USE quizdatabase1;

-- Create a table for students
CREATE TABLE IF NOT EXISTS students (
    student_id VARCHAR(50) PRIMARY KEY,
    student_name VARCHAR(100) NOT NULL,
    student_year VARCHAR (50) NOT NULL,
    student_department VARCHAR(50),
    student_password VARCHAR(50) NOT NULL
);

-- Create a table for teachers
CREATE TABLE IF NOT EXISTS teachers (
    teacher_id VARCHAR(50) PRIMARY KEY,
    teacher_name VARCHAR(100) NOT NULL,
    password VARCHAR(50) NOT NULL
);

CREATE TABLE IF NOT EXISTS questions (
    question_id INT AUTO_INCREMENT PRIMARY KEY,
    subject VARCHAR(255) NOT NULL,
    question_text VARCHAR(1000) NOT NULL,
    option_a VARCHAR(255) NOT NULL,
    option_b VARCHAR(255) NOT NULL,
    option_c VARCHAR(255) NOT NULL,
    option_d VARCHAR(255) NOT NULL,
    correct_option VARCHAR(1) NOT NULL
);

-- Create a table for student marks
CREATE TABLE IF NOT EXISTS student_marks (
    student_id VARCHAR(50) PRIMARY KEY,
    marks INT NOT NULL,
    FOREIGN KEY (student_id) REFERENCES students(student_id)
);

-- Create a table for student attendance
CREATE TABLE IF NOT EXISTS student_attendance (
    student_id VARCHAR(50) PRIMARY KEY,
    attendance VARCHAR(10) NOT NULL,
    FOREIGN KEY (student_id) REFERENCES students(student_id)
);
